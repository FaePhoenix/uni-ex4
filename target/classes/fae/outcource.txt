        public void sendEntryFile(String entryFormat, String fileName) throws IOException{
        FileHelper helper = new FileHelper(fileName);

        ArrayList<String> toSend = helper.getContent();
        int numOfLines = toSend.size();

        DataOutputStream dOS = new DataOutputStream(out);

        dOS.writeInt(numOfLines);
        dOS.writeUTF(entryFormat);
        for (String line : toSend) {
            dOS.writeUTF(line);
        }
        dOS.writeUTF("done");
    }
